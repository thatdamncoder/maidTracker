--------------------------------------------- NEW TABLES


CREATE TABLE maids (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  name text NOT NULL,
  salary numeric(10,2),
  max_leaves integer DEFAULT 2,
  is_active boolean DEFAULT true,
  joined_on date DEFAULT CURRENT_DATE,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Create attendance_monthly table
CREATE TABLE attendance_monthly (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  maid_id uuid NOT NULL REFERENCES maids(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  month integer NOT NULL CHECK (month BETWEEN 1 AND 12),
  year integer NOT NULL,
  attendance_bits bit(31) NOT NULL DEFAULT B'0000000000000000000000000000000',
  marked_upto_day integer DEFAULT 0,
  total_absent integer DEFAULT 0,
  updated_at timestamptz DEFAULT now(),
  UNIQUE(maid_id, month, year)
);

-- Create attendance_exceptions table
CREATE TABLE attendance_exceptions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  maid_id uuid NOT NULL REFERENCES maids(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  date date NOT NULL,
  status text NOT NULL CHECK (status IN ('absent', 'half-day', 'overtime')),
  note text,
  created_at timestamptz DEFAULT now(),
  UNIQUE(maid_id, date)
);

-- Create payments table
CREATE TABLE payments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  maid_id uuid NOT NULL REFERENCES maids(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  amount numeric(10,2) NOT NULL,
  payment_date date DEFAULT CURRENT_DATE,
  receipt_image_url text,
  note text,
  created_at timestamptz DEFAULT now()
);
